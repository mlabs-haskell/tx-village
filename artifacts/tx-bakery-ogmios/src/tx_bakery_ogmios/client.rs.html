48pub struct OgmiosClientConfig {
49    #[builder(default = "Url::parse(\"http://127.0.0.1:1337\").unwrap()")]

57impl OgmiosClientConfigBuilder {
58    fn validate(&self) -> std::result::Result<(), String> {
59        if let Some(url) = &self.url {
60            match url.scheme() {
61                "http" | "https" => Ok(()),
62                scheme => Err(format!(
63                    "Url scheme invalid in OgmiosConfig. Expected https/http, but got {}",
64                    scheme,
65                )),
66            }
67        } else {
68            Ok(())
69        }
70    }
71}

73impl OgmiosClientConfig {
74    pub fn get_ws_url(&self) -> Url {
75        let mut url = self.url.clone();
76        // JUSTIFICATION: We ensure that the url's scheme is either http or https,
77        // so this should always success.
78        url.set_scheme("ws").unwrap();
79        url
80    }

82    pub fn get_restful_health_url(&self) -> Url {
83        // JUSTIFICATION: The base url and the path are always valid.
84        self.url.join("health").unwrap()
85    }

94impl OgmiosClient {
95    pub async fn connect(config: OgmiosClientConfig) -> Result<Self> {
96        let giveup_time = chrono::Local::now() + time::Duration::from_secs(config.startup_timeout);
97
98        let base = time::Duration::from_secs(1);
99        let mut attempt = 0;
100        loop {
101            let health = Self::check_health(&config).await;
102            if health
103                .as_ref()
104                .map_or(false, |h| h.network_synchronization == 1.0)
105            {
106                let client = WsClientBuilder::default()
107                    .build(&config.get_ws_url())
108                    .await?;
109                let client = Self { config, client };
110
111                return Ok(client);
112            } else {
113                if chrono::Local::now() > giveup_time {
114                    return match health {
115                        Err(err) => Err(OgmiosError::StartupError(anyhow!(
116                            "health request failed: {:?}",
117                            err
118                        ))),
119                        Ok(health) => Err(OgmiosError::StartupError(anyhow!(
120                            "couldn't sync: {:?}",
121                            health
122                        ))),
123                    };
124                }
125
126                // Simple exponential backoff
127                let wait_duration = base
128                    .checked_mul(2u32.pow(attempt))
129                    .ok_or(OgmiosError::StartupError(anyhow!("cannot wait any longer")))?;
130                tokio::time::sleep(wait_duration).await;
131                attempt += 1;
132            }
133        }
134    }

162    /// Make a request to ogmios JSON RPC
163    /// Ogmios slightly deviates from the JSON RPC standard, so I couldn't use a 3rd party library
164    /// for this
165    async fn request<P, U>(&self, method: &str, params: P) -> Result<U>
166    where
167        U: serde::de::DeserializeOwned + Serialize,
168        P: ToRpcParams + Send,
169    {
170        self.client.request(method, params).await.map_err(|err| {
171            debug!(%err, "Ogmios JSON RPC call error.");
172            OgmiosError::JSONRpcError(err)
173        })
174    }

180            .await?
181            .json::<OgmiosHealth>()
182            .await?)

233    async fn query_utxos_by_addr(
234        &self,
235        address: &Address,
236    ) -> std::result::Result<BTreeMap<TransactionInput, FullTransactionOutput>, ChainQueryError>
237    {
238        debug!(?address, "Query UTxOs by address");
239        let addr: csl::Address = address
240            .with_extra_info(self.config.network.to_network_id())
241            .try_to_csl()
242            .map_err(OgmiosError::TryFromPLAError)?;
243
244        let addr = addr.to_bech32(Some("addr".to_owned())).map_err(|source| {
245            OgmiosError::ConversionError {
246                label: "Address to Bech32".to_string(),
247                source: anyhow!(source),
248            }
249        })?;
250        let params = QueryLedgerStateUtxoByAddressParams {
251            addresses: vec![addr.to_string()],
252        };
253
254        let resp: QueryLedgerStateUtxoResponse =
255            self.request("queryLedgerState/utxo", params).await?;
256
257        decode_query_ledger_state_utxo_response(resp)
258    }

260    async fn query_utxos_by_ref(

273        let resp: QueryLedgerStateUtxoResponse =
274            self.request("queryLedgerState/utxo", params).await?;
275
276        decode_query_ledger_state_utxo_response(resp)
277    }

280impl Submitter for OgmiosClient {
281    /// Evaluate a transaction and return execution budgets for each script
282    async fn evaluate_transaction(
283        &self,
284        tx_builder: &csl::TransactionBuilder,
285        plutus_scripts: &[csl::PlutusScript],
286        redeemers: &[csl::Redeemer],
287    ) -> std::result::Result<BTreeMap<(csl::RedeemerTag, csl::BigNum), csl::ExUnits>, SubmitterError>
288    {
289        let mut tx_builder = tx_builder.clone();
290
291        tx_builder.set_fee(&csl::BigNum::from(0u64));
292
293        let mut witness_set = csl::TransactionWitnessSet::new();
294
295        let mut script_witnesses = csl::PlutusScripts::new();
296
297        plutus_scripts
298            .iter()
299            .for_each(|script| script_witnesses.add(script));
300
301        let mut redeemer_witnesses = csl::Redeemers::new();
302
303        redeemers
304            .iter()
305            .for_each(|redeemer| redeemer_witnesses.add(redeemer));
306
307        witness_set.set_plutus_scripts(&script_witnesses);
308        witness_set.set_redeemers(&redeemer_witnesses);
309
310        let tx_body = tx_builder.build().map_err(|err| {
311            error!(%err, "Transaction builder error.");
312            SubmitterError(anyhow::anyhow!("Transaction builder error: {}", err))
313        })?;
314        let tx = csl::Transaction::new(&tx_body, &witness_set, None);
315
316        debug!("Evaluating transaction");
317        let params = EvaluateTransactionParams {
318            transaction: TransactionCbor { cbor: tx.to_hex() },
319            additional_utxo: Vec::new(),
320        };
321
322        let resp: EvaluateTransactionResponse = self.request("evaluateTransaction", params).await?;
323
324        resp.into_iter()
325            .map(|budgets| {
326                Ok((
327                    (
328                        to_redeemer_tag(&budgets.validator.purpose)?,
329                        csl::BigNum::from(budgets.validator.index),
330                    ),
331                    csl::ExUnits::new(
332                        &csl::BigNum::from(budgets.budget.memory),
333                        &csl::BigNum::from(budgets.budget.cpu),
334                    ),
335                ))
336            })
337            .collect()
338    }

356    async fn await_tx_confirm(
357        &self,
358        tx_hash: &TransactionHash,
359    ) -> std::result::Result<(), SubmitterError> {
360        info!(?tx_hash, "Awaiting transaction confirmation.");
361        let do_wait = || async {
362            loop {
363                let _ = self.acquire_mempool().await?;
364
365                // TODO: hasTransaction returns a false even when there's a transaction in mempool
366                // Bug report: https://github.com/CardanoSolutions/ogmios/issues/376
367                //
368                // let has_tx = self.has_transaction(tx_hash).await?;
369                let mut has_tx = false;
370                while let NextTransactionResponse::TransactionId {
371                    transaction: Some(resp),
372                } = self.next_transaction().await?
373                {
374                    has_tx = has_tx || resp == tx_hash.into();
375
376                    if has_tx {
377                        break;
378                    }
379                }
380
381                if !has_tx {
382                    let _ = self.release_mempool().await?;
383                    return Result::Ok(());
384                }
385            }
386        };
387
388        let mut retry_counter = 0;
389
390        while retry_counter < 5 {
391            match do_wait().await {
392                Ok(_) => return Ok(()),
393                Err(err) => warn!(
394                    "Unable to confirm transaction {:?}: {}, retrying",
395                    tx_hash, err
396                ),
397            }
398
399            retry_counter += 1;
400        }
401
402        Err(SubmitterError(anyhow!(
403            "Unable to confirm transaction {:?}",
404            tx_hash
405        )))
406    }
407}

409fn decode_query_ledger_state_utxo_response(
410    resp: QueryLedgerStateUtxoResponse,
411) -> std::result::Result<BTreeMap<TransactionInput, FullTransactionOutput>, ChainQueryError> {
412    resp.iter()
413        .map(|utxo| {
414            Ok((
415                TransactionInput::try_from(utxo).map_err(|source| {
416                    OgmiosError::ConversionError {
417                        label: "TransactionInput".to_string(),
418                        source: anyhow!(source),
419                    }
420                })?,
421                FullTransactionOutput::try_from(utxo).map_err(|source| {
422                    OgmiosError::ConversionError {
423                        label: "TransactionInput".to_string(),
424                        source: anyhow!(source),
425                    }
426                })?,
427            ))
428        })
429        .collect()
